import tkinter as tk
from tkinter import font, filedialog, messagebox
import  openpyxl
import  os

class AccountingApp(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Personal Accounting Software")
        self.geometry("800x600")

        # ✅ Initialize active_file here
        self.active_file = None

        # Define a custom font for menus
        menu_font = font.Font(family="Arial", size=18, weight="bold")  # change size here

        # Create a menu bar
        menubar = tk.Menu(self, font=menu_font, bg="orange", fg="black")

        # Add spacing before menus to simulate center alignment
        menubar.add_command(label=">>>>>>>>>>>>>>>>>>>>>>>>>>>>>")  # spacer

        # Suppliers menu
        suppliers_menu = tk.Menu(menubar, tearoff=0, font=menu_font, bg="orange", fg="black")
        suppliers_menu.add_command(label="Add Supplier")
        suppliers_menu.add_command(label="View Suppliers")
        menubar.add_cascade(label="       Suppliers", menu=suppliers_menu)

        # Sales menu
        sales_menu = tk.Menu(menubar, tearoff=0, font=menu_font,bg="orange", fg="black")
        sales_menu.add_command(label="New Sale")
        sales_menu.add_command(label="View Sales")
        menubar.add_cascade(label="     Sales", menu=sales_menu)

        # Reports menu
        reports_menu = tk.Menu(menubar, tearoff=0, font=menu_font, bg="orange", fg="black")
        reports_menu.add_command(label="Monthly Report")
        reports_menu.add_command(label="Profit & Loss")
        menubar.add_cascade(label="  Reports  ", menu=reports_menu,)

        # Stock menu
        stock_menu = tk.Menu(menubar, tearoff=0, bg="orange", fg="black", font=("Arial", 16))
        stock_menu.add_command(label="Create New Stock", command=self.create_new_stock)
        stock_menu.add_command(label="Add Item", command=self.add_item)
        stock_menu.add_command(label="View Stock")
        menubar.add_cascade(label="     Stock", menu=stock_menu)

        self.config(menu=menubar)

        # Add spacing before menus to simulate center alignment
        menubar.add_command(label=">>>>>>>>>>>>>>>>>>>>>>>>>>>>>")  # spacer

        # Attach menu bar
        self.config(menu=menubar)

    def create_new_stock(self):
        choice = messagebox.askquestion(
            "Stock Setup",
            "Do you want to create a NEW Excel file?\n\nClick 'Yes' for new, 'No' to browse existing."
        )

        if choice == "yes":
            # Create new Excel file
            if not os.path.exists("data"):
                os.makedirs("data")
            filepath = "data/transactions.xlsx"

            wb = openpyxl.Workbook()
            ws = wb.active
            ws.title = "Stock"
            ws.append(["Item ID", "Item Name", "Category", "Quantity", "Purchase Price", "Selling Price"])
            wb.save(filepath)

            messagebox.showinfo("Success", f"New stock file created at {filepath}")

        else:
            # Browse existing file
            filepath = filedialog.askopenfilename(
                title="Select Existing Stock File",
                filetypes=[("Excel files", "*.xlsx")]
            )
            if filepath:
                messagebox.showinfo("File Selected", f"You selected: {filepath}")
            else:
                messagebox.showwarning("No File", "No file selected!")
    
    def add_item(self):
        if not self.active_file:
            messagebox.showwarning("No File", "Please create or select a stock file first!")
            return

        # Example: add a dummy item (later we’ll build a form)
        wb = openpyxl.load_workbook(self.active_file)
        ws = wb["Stock"]
        ws.append([1, "Sample Item", "Category A", 10, 50, 75])
        wb.save(self.active_file)

        messagebox.showinfo("Success", "Item added to stock!")

    def view_stock(self):
        if not self.active_file:
            messagebox.showwarning("No File", "Please create or select a stock file first!")
            return

        wb = openpyxl.load_workbook(self.active_file)
        ws = wb["Stock"]

        items = []
        for row in ws.iter_rows(min_row=2, values_only=True):
            items.append(row)

        messagebox.showinfo("Stock Items", f"Current stock:\n{items}")

        

if __name__ == "__main__":
    app = AccountingApp()
    app.mainloop()
