import openpyxl

def record_sale(item_name, quantity_sold):
    wb = openpyxl.load_workbook("data/transactions.xlsx")
    stock_sheet = wb["Stock"]
    
    for row in stock_sheet.iter_rows(min_row=2):  # skip header
        if row[0].value == item_name:  # assuming first column is Item Name
            current_qty = row[1].value  # second column is Quantity
            row[1].value = current_qty - quantity_sold
            break
    
    sales_sheet = wb["Sales"]
    sales_sheet.append([item_name, quantity_sold])
    
    wb.save("data/transactions.xlsx")
