import openpyxl

def record_sale(filepath, sale_data):
    wb = openpyxl.load_workbook(filepath)

    # Deduct from stock
    stock_ws = wb["Stock"]
    item_id, qty_sold = sale_data[1], sale_data[2]  # assuming sale_data = [SaleID, ItemID, Qty, Total]
    for row in stock_ws.iter_rows(min_row=2):
        if row[0].value == item_id:
            row[3].value -= qty_sold  # Quantity column
            break

    # Record sale
    if "Sales" not in wb.sheetnames:
        sales_ws = wb.create_sheet("Sales")
        sales_ws.append(["Sale ID", "Item ID", "Quantity Sold", "Total Amount"])
    else:
        sales_ws = wb["Sales"]

    sales_ws.append(sale_data)
    wb.save(filepath)

def get_sales(filepath):
    wb = openpyxl.load_workbook(filepath)

 # ✅ Create Sales sheet if missing
    if "Sales" not in wb.sheetnames:
        ws = wb.create_sheet("Sales")
        ws.append(["Sale ID", "Item ID", "Quantity Sold", "Total Amount"])
        wb.save(filepath)
        return []  # no records yet

    ws = wb["Sales"]
    return [row for row in ws.iter_rows(min_row=2, values_only=True)]
